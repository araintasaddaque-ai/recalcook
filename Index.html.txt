  <script>
    // Hardcoded nutrition data (per 100g)
    const foodDB = {
      oats: { kcal: 389, gramsPerCup: 90 },
      milk: { kcal: 42, gramsPerCup: 240 },
      banana: { kcal: 89, gramsPerCup: 150 },
      egg: { kcal: 155, gramsPerCup: 50 }
    };

    const ingredients = [];

    function addIngredient() {
      const nameInput = document.getElementById('ingredient');
      const qtyInput = document.getElementById('quantity');
      const name = nameInput.value.trim().toLowerCase();
      const qty = parseFloat(qtyInput.value);

      if (!name || !qty || qty <= 0) return alert("Enter valid ingredient and quantity.");

      if (!foodDB[name]) return alert("Ingredient not found in local DB. Try: oats, milk, banana, egg.");

      const grams = foodDB[name].gramsPerCup * qty;
      const calories = (foodDB[name].kcal / 100) * grams;

      ingredients.push({ name, qty, calories });
      renderTable();

      nameInput.value = '';
      qtyInput.value = '';
    }

    function renderTable() {
      const tbody = document.getElementById('ingredientTable');
      tbody.innerHTML = '';

      let total = 0;
      ingredients.forEach(item => {
        total += item.calories;
        tbody.innerHTML += `
          <tr class="border-b">
            <td class="py-1 capitalize">${item.name}</td>
            <td class="py-1 text-center">${item.qty}</td>
            <td class="py-1 text-center">${item.calories.toFixed(0)}</td>
          </tr>
        `;
      });

      const totalDiv = document.getElementById('totals');
      const totalCal = document.getElementById('totalCalories');
      if (ingredients.length > 0) {
        totalDiv.classList.remove('hidden');
        totalCal.textContent = total.toFixed(0);
      } else {
        totalDiv.classList.add('hidden');
      }
    }
  </script>
